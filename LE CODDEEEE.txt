  <script>
    // ============================================
    // CONFIGURATION N8N
    // ============================================
    const CONFIG = {
      // üîß REMPLACE PAR TON URL N8N R√âELLE
      N8N_WEBHOOK_URL: 'https://karimarouchi.app.n8n.cloud/webhook-test/login-test',
      
      // Options
      DEBUG_MODE: true, // Active les logs de d√©bogage
      REDIRECT_URL: 'formulaire-conges.html',
      REDIRECT_DELAY: 1500, // D√©lai avant redirection (ms)
      REQUEST_TIMEOUT: 10000 // Timeout de la requ√™te (ms)
    };

    // ============================================
    // CLASSE PRINCIPALE DE GESTION DU LOGIN
    // ============================================
    class LoginManager {
      constructor() {
        this.form = document.getElementById('loginForm');
        this.btnLogin = document.getElementById('btnLogin');
        this.btnText = document.getElementById('btnText');
        this.loadingSpinner = document.getElementById('loadingSpinner');
        
        // Champs
        this.emailInput = document.getElementById('email');
        this.passwordInput = document.getElementById('password');
        this.rememberMe = document.getElementById('rememberMe');
        
        // Alertes
        this.alertSuccess = document.getElementById('alertSuccess');
        this.alertError = document.getElementById('alertError');
        this.alertInfo = document.getElementById('alertInfo');
        
        // Debug
        this.debugInfo = document.getElementById('debugInfo');
        this.debugContent = document.getElementById('debugContent');
        
        this.init();
      }

      // ============================================
      // INITIALISATION
      // ============================================
      init() {
        this.log('üöÄ Initialisation du syst√®me de login');
        
        // V√©rifier si d√©j√† connect√©
        this.checkExistingSession();
        
        // √âv√©nements
        this.attachEventListeners();
        
        // Charger les credentials sauvegard√©s
        this.loadRememberedCredentials();
        
        this.log('‚úÖ Initialisation termin√©e');
      }

      // ============================================
      // GESTION DES √âV√âNEMENTS
      // ============================================
      attachEventListeners() {
        // Soumission du formulaire
        this.form.addEventListener('submit', (e) => this.handleSubmit(e));
        
        // Auto-remplissage d√©mo
        document.getElementById('demoCredentials')?.addEventListener('click', () => {
          this.fillDemoCredentials();
        });
        
        // Toggle password visibility
        document.getElementById('togglePassword')?.addEventListener('click', () => {
          this.togglePasswordVisibility();
        });
        
        // Mot de passe oubli√©
        document.getElementById('forgotPassword')?.addEventListener('click', (e) => {
          e.preventDefault();
          this.handleForgotPassword();
        });
        
        // Validation en temps r√©el
        this.emailInput.addEventListener('blur', () => this.validateEmail());
        this.passwordInput.addEventListener('blur', () => this.validatePassword());
        
        // Nettoyer les erreurs lors de la saisie
        this.emailInput.addEventListener('input', () => this.clearError('email'));
        this.passwordInput.addEventListener('input', () => this.clearError('password'));
      }

      // ============================================
      // VALIDATION DU FORMULAIRE
      // ============================================
      validateEmail() {
        const email = this.emailInput.value.trim();
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        
        if (!email) {
          this.showFieldError('email', 'L\'email est requis');
          return false;
        }
        
        if (!emailRegex.test(email)) {
          this.showFieldError('email', 'Format d\'email invalide');
          return false;
        }
        
        this.clearError('email');
        return true;
      }

      validatePassword() {
        const password = this.passwordInput.value;
        
        if (!password) {
          this.showFieldError('password', 'Le mot de passe est requis');
          return false;
        }
        
        if (password.length < 3) {
          this.showFieldError('password', 'Le mot de passe doit contenir au moins 3 caract√®res');
          return false;
        }
        
        this.clearError('password');
        return true;
      }

      validateForm() {
        const emailValid = this.validateEmail();
        const passwordValid = this.validatePassword();
        return emailValid && passwordValid;
      }

      showFieldError(field, message) {
        const input = field === 'email' ? this.emailInput : this.passwordInput;
        const errorSpan = document.getElementById(`${field}Error`);
        
        input.classList.add('error');
        errorSpan.textContent = message;
      }

      clearError(field) {
        const input = field === 'email' ? this.emailInput : this.passwordInput;
        const errorSpan = document.getElementById(`${field}Error`);
        
        input.classList.remove('error');
        errorSpan.textContent = '';
      }

      // ============================================
      // SOUMISSION DU FORMULAIRE
      // ============================================
      async handleSubmit(e) {
        e.preventDefault();
        
        this.log('üì§ Tentative de connexion...');
        this.hideAllAlerts();
        
        // Validation
        if (!this.validateForm()) {
          this.log('‚ùå Validation √©chou√©e');
          return;
        }
        
        this.setLoading(true);
        
        const credentials = {
          email: this.emailInput.value.trim(),
          password: this.passwordInput.value
        };
        
        this.log('üîê Credentials:', { email: credentials.email, password: '***' });
        
        try {
          const result = await this.login(credentials);
          
          if (result.success) {
            this.handleLoginSuccess(result);
          } else {
            this.handleLoginFailure(result);
          }
        } catch (error) {
          this.handleLoginError(error);
        } finally {
          this.setLoading(false);
        }
      }

      // ============================================
      // APPEL API N8N
      // ============================================
      async login(credentials) {
        this.log('üåê Envoi de la requ√™te √† n8n...');
        this.log('URL:', CONFIG.N8N_WEBHOOK_URL);
        
        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), CONFIG.REQUEST_TIMEOUT);
        
        try {
          const requestBody = {
            email: credentials.email,
            password: credentials.password,
            timestamp: new Date().toISOString()
          };
          
          this.log('üì¶ Corps de la requ√™te:', requestBody);
          
          const response = await fetch(CONFIG.N8N_WEBHOOK_URL, {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json',
              'Accept': 'application/json'
            },
            body: JSON.stringify(requestBody),
            signal: controller.signal
          });
          
          clearTimeout(timeoutId);
          
          this.log('üì® Statut de la r√©ponse:', response.status);
          this.log('üì® Headers:', Object.fromEntries(response.headers.entries()));
          
          // Lire la r√©ponse
          const responseText = await response.text();
          this.log('üìÑ R√©ponse brute:', responseText);
          
          let data;
          try {
            data = JSON.parse(responseText);
          } catch (parseError) {
            this.log('‚ö†Ô∏è Erreur de parsing JSON:', parseError);
            throw new Error('R√©ponse invalide du serveur');
          }
          
          this.log('üìä Donn√©es pars√©es:', data);
          
          if (!response.ok) {
            throw new Error(data.message || 'Erreur de connexion');
          }
          
          return data;
          
        } catch (error) {
          clearTimeout(timeoutId);
          
          if (error.name === 'AbortError') {
            throw new Error('Timeout: Le serveur met trop de temps √† r√©pondre');
          }
          
          throw error;
        }
      }

      // ============================================
      // GESTION DES R√âSULTATS
      // ============================================
      handleLoginSuccess(data) {
        this.log('‚úÖ Connexion r√©ussie!', data);
        
        this.showAlert('success');
        
        // Sauvegarder la session
        this.saveSession(data);
        
        // Sauvegarder les credentials si "Se souvenir"
        if (this.rememberMe.checked) {
          this.saveRememberedCredentials(data.email);
        }
        
        // Redirection
        setTimeout(() => {
          this.log('üîÑ Redirection vers:', CONFIG.REDIRECT_URL);
          window.location.href = CONFIG.REDIRECT_URL;
        }, CONFIG.REDIRECT_DELAY);
      }

      handleLoginFailure(data) {
        this.log('‚ùå √âchec de connexion:', data);
        
        if (data.message === 'USER_NOT_FOUND') {
          this.alertInfo.textContent = '‚ÑπÔ∏è Aucun compte trouv√© avec cet email';
          this.showAlert('info');
        } else if (data.message === 'INVALID_PASSWORD') {
          this.alertError.textContent = '‚ùå Mot de passe incorrect';
          this.showAlert('error');
        } else {
          this.alertError.textContent = `‚ùå ${data.message || 'Erreur de connexion'}`;
          this.showAlert('error');
        }
      }

      handleLoginError(error) {
        this.log('üí• Erreur critique:', error);
        
        let errorMessage = '‚ùå Erreur de connexion';
        
        if (error.message.includes('Failed to fetch')) {
          errorMessage = '‚ùå Impossible de joindre le serveur. V√©rifiez votre connexion.';
        } else if (error.message.includes('Timeout')) {
          errorMessage = '‚ùå Le serveur met trop de temps √† r√©pondre';
        } else if (error.message) {
          errorMessage = `‚ùå ${error.message}`;
        }
        
        this.alertError.textContent = errorMessage;
        this.showAlert('error');
      }

      // ============================================
      // GESTION DE LA SESSION
      // ============================================
      saveSession(data) {
        const sessionData = {
          email: data.email,
          nom: data.nom || '',
          prenom: data.prenom || '',
          poste: data.poste || '',
          role: data.role || 'user',
          loginTime: new Date().toISOString()
        };
        
        localStorage.setItem('userSession', JSON.stringify(sessionData));
        localStorage.setItem('userAuth', 'true');
        localStorage.setItem('userEmail', data.email);
        localStorage.setItem('userName', `${data.prenom} ${data.nom}`.trim());
        
        this.log('üíæ Session sauvegard√©e:', sessionData);
      }

      checkExistingSession() {
        const isAuth = localStorage.getItem('userAuth') === 'true';
        
        if (isAuth) {
          this.log('üîê Session existante d√©tect√©e');
          this.alertInfo.textContent = '‚ÑπÔ∏è Vous √™tes d√©j√† connect√©. Redirection...';
          this.showAlert('info');
          
          setTimeout(() => {
            window.location.href = CONFIG.REDIRECT_URL;
          }, 1000);
        }
      }

      // ============================================
      // SE SOUVENIR DES IDENTIFIANTS
      // ============================================
      saveRememberedCredentials(email) {
        localStorage.setItem('rememberedEmail', email);
        this.log('üíæ Email sauvegard√© pour reconnexion');
      }

      loadRememberedCredentials() {
        const rememberedEmail = localStorage.getItem('rememberedEmail');
        
        if (rememberedEmail) {
          this.emailInput.value = rememberedEmail;
          this.rememberMe.checked = true;
          this.log('üìß Email charg√©:', rememberedEmail);
        }
      }

      // ============================================
      // UTILITAIRES UI
      // ============================================
      hideAllAlerts() {
        this.alertSuccess.style.display = 'none';
        this.alertError.style.display = 'none';
        this.alertInfo.style.display = 'none';
      }

      showAlert(type) {
        this.hideAllAlerts();
        const alerts = {
          'success': this.alertSuccess,
          'error': this.alertError,
          'info': this.alertInfo
        };
        alerts[type].style.display = 'block';
      }

      setLoading(isLoading) {
        this.btnLogin.disabled = isLoading;
        
        if (isLoading) {
          this.btnText.style.display = 'none';
          this.loadingSpinner.style.display = 'block';
        } else {
          this.btnText.style.display = 'block';
          this.loadingSpinner.style.display = 'none';
        }
      }

      fillDemoCredentials() {
        this.emailInput.value = 'karim.benali@company.com';
        this.passwordInput.value = 'test123';
        this.clearError('email');
        this.clearError('password');
        this.log('üß™ Identifiants de test remplis');
      }

      togglePasswordVisibility() {
        const type = this.passwordInput.type === 'password' ? 'text' : 'password';
        this.passwordInput.type = type;
        
        const icon = type === 'password' ? 'üëÅÔ∏è' : 'üôà';
        document.getElementById('togglePassword').textContent = icon;
      }

      handleForgotPassword() {
        alert('üîë Fonctionnalit√© "Mot de passe oubli√©" √† venir!\n\nPour le moment, contactez votre administrateur RH.');
      }

      // ============================================
      // DEBUG
      // ============================================
      log(message, data = null) {
        if (!CONFIG.DEBUG_MODE) return;
        
        const timestamp = new Date().toLocaleTimeString();
        const logMessage = `[${timestamp}] ${message}`;
        
        console.log(logMessage, data || '');
        
        // Afficher dans l'interface
        if (this.debugInfo && this.debugContent) {
          this.debugInfo.style.display = 'block';
          const debugLine = data 
            ? `${logMessage}\n${JSON.stringify(data, null, 2)}\n\n`
            : `${logMessage}\n\n`;
          
          this.debugContent.textContent += debugLine;
          
          // Auto-scroll
          this.debugContent.scrollTop = this.debugContent.scrollHeight;
        }
      }
    }

    // ============================================
    // D√âMARRAGE DE L'APPLICATION
    // ============================================
    document.addEventListener('DOMContentLoaded', () => {
      console.log('üéØ Application de login RH d√©marr√©e');
      console.log('üîß Mode debug:', CONFIG.DEBUG_MODE);
      console.log('üåê Webhook URL:', CONFIG.N8N_WEBHOOK_URL);
      
      // Initialiser le gestionnaire de login
      window.loginManager = new LoginManager();
    });

    // ============================================
    // GESTION DES ERREURS GLOBALES
    // ============================================
    window.addEventListener('error', (event) => {
